window.addEventListener('message', function(e){ var data = e.data || {}; var type = (data.type||'').toLowerCase(); if(!type) return; console.log('[preview msg]', type, data); try{ if(type==='togglehitbox'){ if(typeof data.on === 'boolean'){ var dbg=document.getElementById('hitbox-debug'); if(dbg){ dbg.classList.toggle('on', data.on); } if(window.__rockyLive) window.__rockyLive.hitboxDebug=!!data.on; } else if(window.toggleHitbox) window.toggleHitbox(); return; } if(type==='showhitbox' || type==='hidehitbox'){ var dbg=document.getElementById('hitbox-debug'); if(dbg){ dbg.classList.toggle('on', type==='showhitbox'); } return; } if(['idle','cuteidle','walk','goodbye','moviegoodbye','knock','mirror','mirrordance','macarena','jazz','happy','wave'].indexOf(type)!==-1){ if(window.RockyScene){ if(type==='walk'){ if(window.RockyScene.perform) window.RockyScene.perform('walk'); } else if(type==='idle' || type==='cuteidle'){ if(window.RockyScene.perform) window.RockyScene.perform('idle'); } else { if(window.RockyScene.performFacingViewer) window.RockyScene.performFacingViewer(type); else if(window.RockyScene.perform) window.RockyScene.perform(type); } } } }catch(err){ console.error('msg handler', err); } });
